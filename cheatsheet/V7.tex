\noindent\textbf{Newton's method applied to KKT} (same example: $\min\ x_1^2+5x_2^2$ s.t.\ $x_1+x_2=4$, starting at $z_0=(0,0,0)$)

\vspace{4pt}

\noindent
\begin{minipage}[t]{0.30\linewidth}

\textbf{Step 1 — residual at $z_0$:}
\[
  r(z_0) = \begin{bmatrix} 2(0)-0 \\ 10(0)-0 \\ 0+0-4 \end{bmatrix} = \begin{bmatrix} 0 \\ 0 \\ -4 \end{bmatrix}
\]

\textbf{Step 2 — Jacobian} (constant here since $r$ is linear):
\[
  J = \begin{bmatrix} 2 & 0 & -1 \\ 0 & 10 & -1 \\ 1 & 1 & 0 \end{bmatrix}
\]

\end{minipage}\hfill
\begin{minipage}[t]{0.64\linewidth}

\textbf{Step 3 — solve} $J\,\Delta z = -r(z_0)$:
\[
  \begin{bmatrix} 2 & 0 & -1 \\ 0 & 10 & -1 \\ 1 & 1 & 0 \end{bmatrix}
  \begin{bmatrix} \Delta x_1 \\ \Delta x_2 \\ \Delta\lambda \end{bmatrix}
  = \begin{bmatrix} 0 \\ 0 \\ 4 \end{bmatrix}
\]

Row 1: $2\Delta x_1 - \Delta\lambda = 0 \;\Rightarrow\; \Delta x_1 = \tfrac{\Delta\lambda}{2}$

Row 2: $10\Delta x_2 - \Delta\lambda = 0 \;\Rightarrow\; \Delta x_2 = \tfrac{\Delta\lambda}{10}$

Row 3: $\Delta x_1 + \Delta x_2 = 4 \;\Rightarrow\; \tfrac{\Delta\lambda}{2} + \tfrac{\Delta\lambda}{10} = 4 \;\Rightarrow\; 6\Delta\lambda = 40 \;\Rightarrow\; \Delta\lambda = \tfrac{20}{3}$

\[
  \Delta x_1 = \frac{10}{3} \qquad \Delta x_2 = \frac{2}{3}
\]

\textbf{Step 4 — update:} $z_1 = z_0 + \Delta z$
\[
  z_1 = \left(\frac{10}{3},\ \frac{2}{3},\ \frac{20}{3}\right) = x^* \quad \checkmark \quad \text{(one step: KKT system is linear)}
\]

\end{minipage}

\textbf{Newton — unconstrained} \quad $p_0 = -H_f(X_0)^{-1}\nabla f(X_0)$

$f(x_1,x_2)=x_1^2+5x_2^2$,\; $X_0=(3,1)^\top$

\[
  \nabla f(x) = \begin{bmatrix}2x_1\\10x_2\end{bmatrix}
  \;\Rightarrow\;
  \nabla f(X_0) = \begin{bmatrix}6\\10\end{bmatrix}
\]

\[
  H_f = \begin{bmatrix}2&0\\0&10\end{bmatrix}
  \;\Rightarrow\;
  H_f^{-1} = \begin{bmatrix}\tfrac{1}{2}&0\\0&\tfrac{1}{10}\end{bmatrix}
\]

\[
  p_0 = -\begin{bmatrix}\tfrac{1}{2}&0\\0&\tfrac{1}{10}\end{bmatrix}
         \begin{bmatrix}6\\10\end{bmatrix}
       = \begin{bmatrix}-3\\-1\end{bmatrix}
\]

\textbf{Line search:} $\hat{X}_1 = \begin{bmatrix}3-3\alpha\\1-\alpha\end{bmatrix}$

\[
  \phi(\alpha) = (3-3\alpha)^2+5(1-\alpha)^2 = 14-28\alpha+14\alpha^2
\]

\[
  \frac{d\phi}{d\alpha}=-28+28\alpha=0 \;\Rightarrow\; \alpha_0=1
\]

\textbf{Update:} $ 
  X_1 = \begin{bmatrix}3\\1\end{bmatrix}+1\cdot\begin{bmatrix}-3\\-1\end{bmatrix}
      = \begin{bmatrix}0\\0\end{bmatrix}
$

$f(X_1)=0$ — one step, exact (quadratic $f$, $\alpha=1$ always)

\end{minipage}\hfill
\begin{minipage}[t]{0.46\linewidth}
\textbf{BFGS quasi-Newton} \quad $p_0 = -\tilde{\nabla}^2 f(X_0)^{-1}\nabla f(X_0)$,\; $B_0=I$

$f(x_1,x_2)=x_1^2+5x_2^2$,\; $X_0=(3,1)^\top$

\[
  \nabla f(x)=\begin{bmatrix}2x_1\\10x_2\end{bmatrix}
  \;\Rightarrow\;
  \nabla f(X_0)=\begin{bmatrix}2\cdot3\\10\cdot1\end{bmatrix}=\begin{bmatrix}6\\10\end{bmatrix}
\]

\textbf{Candidate:} $\hat{X}_1 = X_0+\alpha p_0 = \begin{bmatrix}3\\1\end{bmatrix}-\alpha\begin{bmatrix}6\\10\end{bmatrix} = \begin{bmatrix}3-6\alpha\\1-10\alpha\end{bmatrix}$

\[
  \phi(\alpha)=(3-6\alpha)^2+5(1-10\alpha)^2
\]
\[
  = 9-36\alpha+36\alpha^2+5(1-20\alpha+100\alpha^2)
\]
\[
  = 14-136\alpha+536\alpha^2
\]

\[
  \frac{d\phi}{d\alpha}=-136+1072\alpha=0
  \;\Rightarrow\; \alpha_0=\frac{136}{1072}\approx0.1269
\]

\[
  X_1 = X_0+\alpha_0\nabla f(X_0)
      = \begin{bmatrix}3\\1\end{bmatrix}-0.1269\begin{bmatrix}6\\10\end{bmatrix}
      = \begin{bmatrix}2.239\\-0.269\end{bmatrix}
\]

\textbf{BFGS vectors:}
\[
  s_0 = -\alpha_0\nabla f(X_0) = -0.1269\begin{bmatrix}6\\10\end{bmatrix}
      = \begin{bmatrix}-0.7614\\-1.269\end{bmatrix}
\]
\[
  y_0 = \nabla f(X_1)-\nabla f(X_0)
      = \begin{bmatrix}2\cdot2.239\\10\cdot(-0.269)\end{bmatrix}-\begin{bmatrix}6\\10\end{bmatrix}
      = \begin{bmatrix}-1.522\\-12.689\end{bmatrix}
\]

\end{minipage}
